    @GetMapping("employee")

        return employees;
    }




    @PostMapping("employee/add")
    public String  postEmployee(@RequestBody String employeeName){
        employees.add(employeeName);
        return "Employee added";
    }




    @PutMapping("employee/update/{index}")
    public String  putEmployee(@PathVariable Integer index, @RequestBody String employeeName){
        employees.set(index,employeeName);

    }
    @DeleteMapping("employee/delete/{index}")
    public String deleteBook(@PathVariable Integer index){
        employees.remove(index);

    public class foodNameController {
    private ArrayList<String> food = new ArrayList<>();
    @GetMapping("food")
    public ArrayList<String> getFoodNames(){

        return food;
    }
    @PostMapping("food/add")
    public String  postFoodNames(@RequestBody String foodNames){
        food.add(foodNames);
        return "food added";
    }
    @PutMapping("food/update/{index}")
    public String  putFood(@PathVariable Integer index, @RequestBody String foodNames){
        food.set(index,foodNames);
        return "food name updated";
    }
    @DeleteMapping("food/delete/{index}")//dynamic route,
    public String deleteFoodNames(@PathVariable Integer index){
        food.remove(index);
        return "food deleted";
    }
